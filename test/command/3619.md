```
% pandoc -f html -t markdown --reference-links --markdown-headings=setext
<a href="foo">bar</a>: baz
^D
[bar][]: baz

  [bar]: foo
```

```
% pandoc -f html -t markdown --reference-links --markdown-headings=setext
<a href="foo">bar</a>(baz)
^D
[bar][](baz)

  [bar]: foo
```

```
% pandoc -f html -t markdown_strict --reference-links --markdown-headings=setext
<a href="a">foo</a><br/><a href="b">bar</a>
^D
[foo][]  
[bar]

  [foo]: a
  [bar]: b
```
