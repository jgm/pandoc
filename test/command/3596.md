% pandoc -f html -t markdown-raw_html-bracketed_spans-native_spans

</ul>

```
% pandoc -f html -t markdown-raw_html-bracketed_spans-native_spans
<ul>
<li>foo</li>
<li id="id">bar<ul><li>subbar</li></ul></li>
<li>baz</li>
</ul>
^D
-   foo
-   bar
    -   subbar
-   baz
```


```
% pandoc -f html -t markdown
<ul>
<li>foo</li>
<li id="id">bar</li>
<li>baz</li>
</ul>
^D
-   foo
-   [bar]{#id}
-   baz
```


```
% pandoc -f html -t markdown
<ul>
<li><p>foo</p></li>
<li id="id"><p>bar</p></li>
<li><p>baz</p></li>
</ul>
^D
-   foo

-   ::: {#id}
    bar
    :::

-   baz

```
